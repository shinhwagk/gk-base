#####对于一个循环使用函数式风格的演变
```scala
def printArgs(args: Array[String]):Unit={
	var i = 0
    while(i < args.length){
    	println(args(i))
        i += 1
    }
}

def printArgs(args: Array[String]):Unit={
	for(arg <= args)
    	println(arg)
}

def printArgs(args: Array[String]):Unit={{
	args.foreach(println)
}
```
